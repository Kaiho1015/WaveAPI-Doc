异步处理模式 — 提交请求后返回任务 ID，通过 [查询任务状态](/cn/api-reference/task/status) 接口获取生成结果。视频链接有效期 24 小时。

<RequestExample>
```bash cURL
curl -X POST https://qingbo.dev/v1/tasks \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "model": "wan2.5-t2v",
    "action": "generate",
    "prompt": "春天的樱花树下，花瓣随风飘落",
    "aspect_ratio": "16:9",
    "duration": 5
  }'
```

```python Python
import requests

response = requests.post(
    "https://qingbo.dev/v1/tasks",
    headers={"Authorization": "Bearer YOUR_API_KEY"},
    json={
        "model": "wan2.5-t2v",
        "action": "generate",
        "prompt": "春天的樱花树下，花瓣随风飘落",
        "aspect_ratio": "16:9",
        "duration": 5
    }
)

task_id = response.json()["task_id"]
print(f"任务已提交: {task_id}")
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io"
    "net/http"
)

func main() {
    payload := map[string]interface{}{
        "model":        "wan2.5-t2v",
        "action":       "generate",
        "prompt":       "春天的樱花树下，花瓣随风飘落",
        "aspect_ratio": "16:9",
        "duration":     5,
    }

    body, _ := json.Marshal(payload)
    req, _ := http.NewRequest("POST", "https://qingbo.dev/v1/tasks", bytes.NewBuffer(body))
    req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    resp, _ := http.DefaultClient.Do(req)
    defer resp.Body.Close()
    result, _ := io.ReadAll(resp.Body)
    fmt.Println(string(result))
}
```

```java Java
import java.net.http.*;
import java.net.URI;

public class Main {
    public static void main(String[] args) throws Exception {
        String payload = """
        {
          "model": "wan2.5-t2v",
          "action": "generate",
          "prompt": "春天的樱花树下，花瓣随风飘落",
          "aspect_ratio": "16:9",
          "duration": 5
        }
        """;

        HttpClient client = HttpClient.newHttpClient();
        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create("https://qingbo.dev/v1/tasks"))
            .header("Authorization", "Bearer YOUR_API_KEY")
            .header("Content-Type", "application/json")
            .POST(HttpRequest.BodyPublishers.ofString(payload))
            .build();

        HttpResponse<String> response = client.send(request,
            HttpResponse.BodyHandlers.ofString());
        System.out.println(response.body());
    }
}
```

## 可用模型

| 模型 ID | 说明 |
|---------|------|
| `wan2.5-t2v` | 万象 2.5 文生视频 |
| `wan2.5-i2v` | 万象 2.5 图生视频 |
| `wan2.2-i2v-flash` | 万象 2.2 图生视频快速版 |
| `wan2.2-i2v-plus` | 万象 2.2 图生视频增强版 |

## 统一参数

<ParamField body="model" type="string" required>
  模型 ID
</ParamField>

<ParamField body="action" type="string" default="generate">
  操作类型
</ParamField>

<ParamField body="prompt" type="string" required>
  视频描述文本
</ParamField>

<ParamField body="aspect_ratio" type="string" default="16:9">
  宽高比，后台自动映射为最佳分辨率
</ParamField>

<ParamField body="duration" type="integer" default="5">
  视频时长（秒）
</ParamField>

<ParamField body="resolution" type="string">
  分辨率：`480p`、`720p`、`1080p`
</ParamField>

<ParamField body="seed" type="integer">
  随机种子，范围 0-2147483647
</ParamField>

<ParamField body="watermark" type="boolean" default="false">
  是否添加水印
</ParamField>

<ParamField body="image_urls" type="string[]">
  参考图片 URL（图生视频模型使用）
</ParamField>

<ParamField body="first_frame" type="string">
  首帧图片 URL
</ParamField>

<ParamField body="last_frame" type="string">
  尾帧图片 URL
</ParamField>

<ParamField body="negative_prompt" type="string">
  不希望出现的内容，最多 500 字符
</ParamField>

<ParamField body="callback_url" type="string">
  Webhook 回调地址
</ParamField>

<ParamField body="extra" type="object">
  透传参数
</ParamField>

## 生成模式

| 模式 | 模型 | 参数 |
|------|------|------|
| 文生视频 | `wan2.5-t2v` | `prompt` |
| 图生视频 | `wan2.5-i2v` / `wan2.2-i2v-*` | `prompt` + `image_urls` |
| 首尾帧 | `wan2.5-i2v` | `first_frame` + `last_frame` |
